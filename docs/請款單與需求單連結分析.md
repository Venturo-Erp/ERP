## 🎯 **團確單的付款狀態顯示**

### 邏輯

```typescript
// 檢查需求單是否有請款單（已付款）
async function getPaymentStatus(requestId: string) {
  const { data: items } = await supabase
    .from('payment_request_items')
    .select('*, payment_requests!inner(status, paid_at)')
    .eq('tour_request_id', requestId)
  
  if (!items || items.length === 0) {
    return { status: 'unpaid', label: '未付款' }
  }
  
  const hasPaid = items.some(item => 
    item.payment_requests.paid_at != null
  )
  
  return hasPaid 
    ? { status: 'paid', label: '已付清', paidAt: items[0].payment_requests.paid_at }
    : { status: 'pending', label: '待付款' }
}
```

```tsx

// PaymentStatusBadge 組件
function PaymentStatusBadge({ requestId }) {
  const status = usePaymentStatus(requestId)
  
  return (
    <span className={cn(
      'px-2 py-1 rounded text-xs',
      status.status === 'paid' && 'bg-green-100 text-green-700',
      status.status === 'pending' && 'bg-yellow-100 text-yellow-700',
      status.status === 'unpaid' && 'bg-gray-100 text-gray-700'
    )}>
      {status.label}
    </span>
  )
}
```
